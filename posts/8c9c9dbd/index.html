通过对 .Net6 和 .Net7 反射性能的简要测试，为性能优化提供依据。本文测试了以下项目：

名称内容
Public直接访问公开属性
Private通过反射访问私有属性
CachedPrivate通过缓存的反射访问私有属性
DelegatePrivate通过委托访问私有属性
CachedDelegatePrivate通过缓存的委托访问私有属性
ExpressionPrivate通过表达式树访问私有属性
CachedExpressionPrivate通过缓存的表达式树访问私有属性

using System.Linq.Expressions;
using System.Reflection;
using BenchmarkDotNet.Attributes;
using BenchmarkDotNet.Jobs;

namespace ConsoleApp;

[SimpleJob(RuntimeMoniker.Net70)]
[SimpleJob(RuntimeMoniker.Net60)]
[MemoryDiagnoser]
public class ReflectionBenchmark
{
    static readonly User _user = new();

    [Benchmark]
    public string Public()
    {
        return _user.Pub;
    }

    [Benchmark]
    public string Private()
    {
        var propertyInfo = typeof(User).GetProperty("Pri", BindingFlags.Instance | BindingFlags.NonPublic)!;
        return (string)propertyInfo.GetValue(_user)!;
    }

    static readonly PropertyInfo _propertyInfo = typeof(User).GetProperty("Pri", BindingFlags.Instance | BindingFlags.NonPublic)!;

    [Benchmark]
    public string CachedPrivate()
    {
        return (string)_propertyInfo.GetValue(_user)!;
    }

    [Benchmark]
    public string DelegatePrivate()
    {
        var propertyInfo = typeof(User).GetProperty("Pri", BindingFlags.Instance | BindingFlags.NonPublic)!;
        var delFunc = (Func<User, string>)Delegate.CreateDelegate(typeof(Func<User, string>), propertyInfo.GetMethod!);
        return delFunc(_user);
    }

    static readonly Func<User, string> _delFunc = (Func<User, string>)Delegate.CreateDelegate(typeof(Func<User, string>), _propertyInfo.GetMethod!);

    [Benchmark]
    public string CachedDelegatePrivate()
    {
        return _delFunc(_user);
    }

    [Benchmark]
    public string ExpressionPrivate()
    {
        var propertyInfo = typeof(User).GetProperty("Pri", BindingFlags.Instance | BindingFlags.NonPublic)!;
        var parameter = Expression.Parameter(typeof(User));
        var propertyExpression = Expression.Property(parameter, propertyInfo);
        var expFunc = Expression.Lambda<Func<User, string>>(propertyExpression, parameter).Compile();
        return _delFunc(_user);
    }

    static readonly ParameterExpression _parameter = Expression.Parameter(typeof(User));
    static readonly MemberExpression _propertyExpression = Expression.Property(_parameter, _propertyInfo);
    static readonly Func<User, string> _expFunc = Expression.Lambda<Func<User, string>>(_propertyExpression, _parameter).Compile();

    [Benchmark]
    public string CachedExpressionPrivate()
    {
        return _expFunc(_user);
    }
}

public sealed class User
{
    public string Pub => "Public";
    private string Pri => "Private";
}

简单总结如下：

  • 通过表达式树访问属性，除直接访问属性外，其性能最佳，但是开销较大，且编写繁琐。
  • 通过反射访问属性，在 .Net7 中优化显著，同时利用缓存也可以获得较好的性能。
  • 通过委托访问属性，其性能和开销较为均衡，适合大多数性能优化使用。
